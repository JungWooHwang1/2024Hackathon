# 대화 기록에 새로운 대화를 추가하는 함수
def add_to_history(new_text: str):

# 대화 기록을 기반으로 프롬프트를 생성하는 함수
def generate_prompt():
    base_prompt = '''
1. 당신은 친절하고 전문적인 의사입니다. 모든 분야에서 전문 지식을 갖추고 있으며, 환자들에게 철저하고 이해하기 쉬운 설명을 제공하는 데 시간을 할애합니다. 
2. 오늘은 다양한 건강 문제로 병원을 찾은 새로운 환자를 만나게 됩니다. 환자는 소화기 문제, 만성 통증, 정신 건강 문제, 피부 질환 등 여러 가지 증상을 호소하고 있습니다.

5. 사용자가 한 질문을 절대로 되물으면 안됩니다 무조건 사용자가 질문한 내용을 최대한 분석하여 솔루션을 제공하세요.

사용자 질문 : 
    '''
    history_prompt = "\n".join(conversation_history)
    return base_prompt + "\n\n" + history_prompt

# 사용자의 입력을 받아 챗봇의 응답을 생성하는 함수
def chat_with_doctor(user_input: str):
    # 대화 기록에 사용자 입력 추가
    add_to_history(f"사용자: {user_input}")
    
    # 프롬프트 생성
    prompt = generate_prompt()
    
    # 모델로부터 응답 생성
    response = model.generate_content(prompt, stream=True)
    response.resolve()
    
    # 대화 기록에 챗봇 응답 추가
    add_to_history(f"의사: {response.text}")
    
    return response.text

@app.route('/api/forecast', methods=['GET'])
def get_forecast():
    # 클라이언트로부터 받는 파라미터
    dissCd = request.args.get('dissCd')
    znCd = request.args.get('znCd')
    lowrnkZnCd = request.args.get('lowrnkZnCd')

    # API 요청 파라미터 설정
    params = {
        'serviceKey': serviceKey,
        'numOfRows': '100',
        'pageNo': '1',
        'type': 'json',
        'dissCd': dissCd,
        'znCd': znCd
    }
    # API 호출

        try:
            data = response.json()

            filtered_items = []

            for item in items:
                if item.get('lowrnkZnCd') == lowrnkZnCd:
                    filtered_items.append({
                        "일자": item.get('dt'),
                        "지역명": region_codes.get(znCd, "Unknown Region") + " " + subregion_codes.get(lowrnkZnCd, "Unknown Subregion"),

                        "예측진료건수": item.get('cnt'),
                        "예측위험도": item.get('risk'),
                        "위험도 지침": item.get('dissRiskXpln')
                    })

            return jsonify(filtered_items)

        except json.JSONDecodeError:
            return jsonify({"error": "JSON 디코딩 오류가 발생했습니다."}), 500
    else:
        return jsonify({"error": f"요청 실패: 상태 코드 {response.status_code}"}), response.status_code
    



def get_existing_dates(region_code):
    connection = get_db_connection()
    cursor = connection.cursor()

    table_name = f"forecast_{region_code}"
    query = f"SELECT dt FROM {table_name};"
    cursor.execute(query)
    existing_dates = cursor.fetchall()

    connection.commit()
    cursor.close()
    connection.close()

    # 결과를 리스트 형태로 변환
    return [date[0] for date in existing_dates]

def insert_forecast_data(region_code, data):
    connection = get_db_connection()
    cursor = connection.cursor()

    table_name = f"forecast_{region_code}"
    insert_query = f"""
    INSERT INTO {table_name} (dt, region_name, subregion_name, disease_name, cnt, risk, risk_explanation)
    VALUES (%s, %s, %s, %s, %s, %s, %s);
    """
    for item in data:
        cursor.execute(insert_query, (
            item["일자"],
            item["지역명"],
            item["질병명"],
            item["예측진료건수"],
            item["예측위험도"],
            item["위험도 지침"]
        ))

    connection.commit()
    cursor.close()
    connection.close()
